Создаем постоянную точку монтирования

```shell
sudo mkdir -p /mnt/linux-backup  # Создаем папку
sudo chown $USER:$USER /mnt/linux-backup  # Даем права текущему пользователю
```

---

Команды для монтирования и размонтирования:

```shell
sudo mount /dev/sda3 /mnt/linux-backup
sudo umount /mnt/linux-backup
```

Алиасы:

```shell
alias mount_backup='sudo mount /dev/sda3 /mnt/linux-backup'
alias umount_backup='sudo umount /mnt/linux-backup'
```

---

```shell
alias sync 'mount_backup && rsync -av --ignore-existing /home/vitaly/Документы/ /run/media/vitaly/linux-backup/myBackup/ && umount_backup'
```

### shell-скрипт

Лучше всего реализовать через shell-скрипт + служба [systemd](obsidian://open?vault=obsidian&file=%D0%9A%D0%B0%D0%BA%20%D0%BD%D0%B0%D0%BF%D0%B8%D1%81%D0%B0%D1%82%D1%8C%20%D1%81%D0%B2%D0%BE%D1%8E%20%D1%81%D0%BB%D1%83%D0%B6%D0%B1%D1%83%20unit%20systemd).

shell-скрипт:
```shell
#!/usr/bin/bash

  DEV=/dev/sda3
  MNT=/mnt/linux-backup
  SRC=/home/vitaly/Документы/
  DST=/mnt/linux-backup/myBackup/

  # Монтирование
  if mountpoint -q "$MNT"; then
      echo "Диск уже смонтирован"
  else
      echo "Монтирование..."
      sudo mount "$DEV" "$MNT"
  fi

  # синхронизация
  rsync -av --ignore-existing "$SRC" "$DST" # мои документы
  rsync -av --ignore-existing /home/vitaly/scripts "$DST"  # мои скрипты
  rsync -av --ignore-existing "/home/vitaly/PortProton/data/prefixes/DEFAULT/drive_c/users/vitaly/Documents/The Witcher 3" "$DST"  # сохранения ведьмак 3
  echo "✅ rsync завершён"

  # Размонтирование
  sudo umount "$MNT"
  if mountpoint -q "$MNT"; then
      echo "❌ Диск не размонтирован"
  else
      echo "✅ Диск размонтирован"
  fi

```
